+#include <linux/kernel.h>
+#include <linux/module.h>
+#include <linux/sched.h>
+#include <linux/string.h>
+#include <linux/syscalls.h>
+#include <linux/fcntl.h>
+#include <asm/uaccess.h>
+#include <uapi/asm-generic/errno.h>
+#include "rtNice.h"
+
+long rt_nice(int pid, int soft_rq)
+{
+	
+	int flag=0;
+	long finalTime=(long) soft_rq*1000000000; 
+	struct task_struct *task;
+	long arg= (long)pid;
+
+	if(pid <= 0)
+	{
+		printk("PID cannot be less than 1");
+		return -EINVAL;
+	}
+
+	if(soft_rq<0)
+	{
+		printk("Soft requirement cannot be negative");
+		return -EINVAL;
+	}
+
+	for_each_process(task)
+	{
+		if((long)task_pid_nr(task)==arg)
+		{
+			flag=1;
+			printk("Changed rt_nice value of %s to %ld\n",task->comm,finalTime);
+			task->se.rt_nice = finalTime;
+		}
+	}
+
+	if(flag==0)
+	{
+		printk("No such process found\n");
+		return -ESRCH;
+	}
+	
+	return 0;
+}
+
+SYSCALL_DEFINE2(rt_nice, int, pid, int, soft_rq)
+{
+	return rt_nice(pid, soft_rq);
+}
